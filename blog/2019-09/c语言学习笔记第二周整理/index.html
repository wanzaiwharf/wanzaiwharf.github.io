课后作业

第一题

计算1到5的乘积，即5的阶乘

#include<stdio.h>

int main()
{
	int product;
	product=1*2*3*4*5;
	printf("The product from 1 to 5 is %d",product);
	return 0;
}

第二题

手动输入n，然后计算从1依次相乘到n的乘积

#include<stdio.h>

int main()
{
	int prod(int n);
	int product;
	int n;
	scanf("%d",&n);
	product=prod(n);
	printf("The product from 1 to %d is %d\n",n,product);
	return 0;
}

int prod(int n)
{
	int p,i;
	p=1;
	i=1;
	do{
		p=p*i;
		i=i+1;
	}while(i<=n);//do while循环
	return p;
}

第三题

在第二题的基础上更改程序，使其能计算从1到n的奇数依次相乘的乘积

#include<stdio.h>

int main()
{
	int prod(int n);
	int product;
	int n;
	scanf("%d",&n);
	product=prod(n);
	printf("The product from 1 to %d is %d\n",n,product);
	return 0;
}

int prod(int n)
{
	int p,i;
	p=1;
	i=1;
	do{
		p=p*i;
		i=i+2;//改一个加2即可，主要需要熟练掌握循环的逻辑
	}while(i<=n);
	return p;
}

第四题

编写一个将华氏度转换为摄氏度的程序

#include<stdio.h>

int main()
{
    int fahr;
    for(fahr=0;fahr<=300;fahr=fahr+20)
        printf("%d %6.1f\n",fahr,(2.0/9.0)*(fahr-32));
    return 0;
}

输入一个年份计算其是否为润年

#include<stdio.h>
    #define LOWER 2000
    #define UPPER 2100//初始化参数
    
    int main()
    {
        int isLeapYear(int year);

        do
        {
            if(isLeapYear(year)==1)//==表示是否相等逻辑判断中的相等，与之相对的!=表示不等
                printf("%d is not leap year\n",year);
            else
                printf("%d is not leap year\n",year);//if else语句
            year=year+1;
        }while(year<=UPPER);
        return 0;
    }
    
    int isLeapYear(int year)
    {
        int r;
        r=year%4;//%表示求余计算r为余数
        if(r!=0)
        {
            return 1;
        }else{
            r=year%400;
            if(r==1)
            {
                return 1;
            }else{
                return 0;
            }

第六题

输入n计算 $$ 1*1⁄2*1⁄3*……*1/n $$

#include<stdio.h>
int main()
{
    float sign=1;
    float sum=0;
    float deno=1;
    float term;
    do
    {
        term=sign*(1/deno);
        sum=sum+term;
        sign=-1*sign;
        deno=deno+1;
    }
    printf("sum = %f\n",sum);
    return 0;
}
