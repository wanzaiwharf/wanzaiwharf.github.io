课后习题

第一题

输入一串字符，并进行回显

#include<stdio.h>

int main()
{
    int c;
    c=getchar();//getchar 读取键盘输入并将读取到的值放入变量中
    while(c!=EOF)
    {
        putchar(c);//循环中的结束条件EOF表示ctrl+d或回车
        c=getchar();
    }
    return 0;
}

进阶，要求同时统计单词个数，空格个数，回车个数

#include<stdio.h>

int main()
{

            if(state==0)

            state=1;
            }
        }

第五题

统计每个数字、空白符、以及其它字符出现的次数

#include<stdio.h>

int main()
{
    int c,i,nwhite,nother;
    int ndigit[10];//数组初接触
    nwihte=nother=0;
    for(i=0;i<10;++i)
        ndigit[i]=0;
    
    while((c=getchar())!=EOF)
        if(c>='0'&&c<='9')
            ++ndigit[c-'0'];
    	else if(c==' '||c=='\n'||c=='\t')
        	++nwhite;
    	else
        	++nother;
    printf("\ndigits =");
    for(i=0;i<10;++i)
        printf(" %d",ndigit[i]);

第六题

#include<stdio.h>
int power(int m,int n);

int main()
{
    int i;
    for(i=0;i<10;i++)
        printf("%3d %6d %6d\n",i,power(2,i),power(-3,i));
    return 0;
}

int power(int base,int n)
{
    int i,p;
    p=1;
    for(i=1;i<=n;++i)
        p=p*base;
    return p;
}

求n，m的最大公约数和最小公倍数

#include<stdio.h>

int main()
{
    int m,n,a,b,x,y,z;
    printf("输入数据m和n，中间用空格隔开\n");
    scanf("%d %d",&m,&n);
    a=m;
    b=n;
	while(a%b!=0)
    {
        x=a%b;
        a=b;
        b=x;
    }
    printf("m,n的最大公约数为%d\n",x);//求最大公约数部分
    y=m*n;
    z=y/x;
    printf("m,n的最小公倍数为%d\n",z);//求最小公倍数部分
    return 0;        
}

第三题

更改单词首字母为大写，其余更改为小写

#include <stdio.h>

int main()
{
    char a[100];
    int i;
    scanf("%s",a);
    for(i=0;a[i]!='\0';i++){
        if(a[i]>='A'&&a[i]<='Z')//大写字母 
            a[i]+=32;
        else if(a[i]>='a'&&a[i]<='z')//小写字母 
            a[i]-=32;
            }
		   
		if(a[0]>='a'&&a[0]<='z')//小写字母 
            a[0]-=32;
    puts(a);
    return 0;
}

第四题

输入整数n，计算 $$ 1+1/1！+1/2！+……1/n！ $$

#include<stdio.h>

int main()
{
    int i,n,j;
    float s,p;
    s=0;
    i=1;
    printf("输入大于1的整数n,输入完后按回车开始计算\n");
    scanf("%d",&n);
    while(i<=n)
    {
        p=1;
        j=1;//初始化分母的计算
        while(j<=i)//计算分母
        {
            p=p*j;
        	j++;
        }
        s=s+1/p;
        i++;
    }
    printf("计算结果为%d\n",s);
    return 0;
}
