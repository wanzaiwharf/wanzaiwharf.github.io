例1

#include<stdio.h>
#define MAXLINE 1000

int mygetline(char line[],int maxline);
void copy(char to[],char from[]);

int main()
{
    int len,max;
	char longest[MAXLINE];
    char line[MAXLINE];
    max=0;
    while((len=mygetline(line,MAXLINE))>0)
        if(len>max)
        {
            max=len;
            copy(longest,line);
        }
    if(max>0)
        printf("\nThe longest is: %s\n length=%d\n",longest,max);
    return 0;
}

int mygetline(char s[],int lim)
{
    int c,i;
    for(i=0;i<lim-1 && (c=getchar())!=EOF && c!='\n';++i)
        s[i]=c;
    if(c=='\n')
    {
        s[i]=c;
        ++i;
    }
    s[i]='\0';
    return i;
}

void copy(char to[],char from[])
{
    int i;
    i=0;
    while((to[i]=from[i])!='\0')
        ++i;
}

例3 统计一个字符串的字符个数

#include<stdio.h>
#define MAXLENGTH 1000
int mystrlen(char s[]);

int main()
{
    char mystring[MAXLENGTH];
    int len;
    scanf("%s",mystring);
    len=mystrlen(mystring);
    printf("Len=%d\n",len);
    return 0;
}

int mystrlen(char s[])
{
    int i;
    i=0;
    while(s[i]!='\0')
        ++i;
    return i;
}

例4 枚举常量的值

#include<stdio.h>

enum escapes
{
BELL='\a',
BACKSPACE='\b',
TAB='\t',
NEWLINE='\n',
VTAB='\v',
RETURN='\r',
};

int main()
{
    enum escapes be,ba,ta,ne,vt,re;
    be=BELL;
    ba=BACKSPACE;
    ta=TAB;
    ne=NEWLINE;
    vt=VTAB;
    re=RETURN;
    printf("be=%d\n",be);
    printf("ba=%d\n",ba);
    printf("ta=%d\n",ta);
    printf("ne=%d\n",ne);
    printf("vt=%d\n",vt);
    printf("re=%d\n",re);
    return 0;
}

例6 逻辑运算符中求值顺序

#include<stdio.h>

int main()
{
    int i,j;
    i=j=0;
    if(++i==1||++j>0)
    {
        printf("i=%d,j=%d\n",i,j);
    }
    i=j=0;
    if(++i!=1||++j>0)
    {
        printf("i=%d,j=%d\n",i,j);
    }
    return 0;
}

例7

7.1 将字符串转换成整数

#include<stdio.h>
#define MAXLEN 100

int atoi(char[]);

int main()
{
    char s[MAXLEN];
    int data;
    scanf("%s",s);
    data=atoi(s);
    printf("%d\n",data);
    return 0;
}

int atoi(char s[])
{
    int i,n;
    n=0;
    for(i=0;s[i]>='0'&&s[i]<='9';++i)
        n=10*n+(s[i]-'0');
    return n;
}

7.2 将大写字母转化成小写字母

#include<stdio.h>

int lower(int);

int main()
{
    int uc,lc;
    uc=getchar();
    lc=lower(uc);
    printf("lc=%c\n",lc);
    return 0;
}

int lower(int c)
{
    if(c>='A'&&c<='Z')
        return c-'A'+'a';
    else
        return c;
}

例9 强制类型转换

10.1

10.2

#include<stdio.h>

int main()
{
    int a=5,b=10,sum=0;
    sum=sum+ ++a;
    printf("a=%d b=%d sum=%d\n",a,b,sum);
    sum=sum+ b++ + ++a;
    printf("a=%d b=%d sum=%d\n",a,b,sum);
    sum=sum+ --b + --a;
    printf("a=%d b=%d sum=%d\n",a,b,sum);
    sum=sum+ --b + ++a;
    printf("a=%d b=%d sum=%d\n",a,b,sum);
}

#include<stdio.h>

int main()
{
    int a,b,c,d;
    a=b=c=0;
    if(a++>1||b++<1||c++==1)
        d=a+b+c++;
    printf("a=%d b=%d c=%d d=%d\n",a,b,c,d);

}
