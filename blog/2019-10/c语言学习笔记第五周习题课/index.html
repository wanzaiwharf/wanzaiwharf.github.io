第一题

编写程序输出所有水仙花数

水仙花数：一个三位数，各位数字的立方和等于该数本身

#include<stdio.h>

int main()
{
    int hun, ten, ind, n;
    printf("全部水仙花数如下:\n");
    for( n=100; n<1000; n++ )  /*的取值范围*/
    {
        hun = n / 100;
        ten = (n-hun*100) / 10;
        ind = n % 10;
        if(n == hun*hun*hun + ten*ten*ten + ind*ind*ind)  
            /*各位上的立方和是否与原数n相等*/
            printf("%d  ", n);
    }
    printf("\n");
    return 0;
}

第二题

编写程序输出1000以内的完数

完数：除自身以外所有因子之和等于自身的数

#include<stdio.h>

int main()
{
    int j,s,n;
    for(n=2;n<1000;n++)
    {
        for( j=1; j<n; j++ )
        {
            if(n%j == 0)  /*判断j是否为i的因子*/
                s += j;
        }
         if(s == n)  /*判断因子之和是否和原数相等*/
         printf("%d 是1000以内的完数\n", n);
    }
    return 0;
}

要求主函数中读入，主函数中输出

读入一组文本行（行数大于2），打印最短的两个文本行

#include <stdio.h>
#define MAXLINE 1000

int readline(char line[], int minline);
void copy(char to[], char from[]);

int main(void)
{
    int len;
    int min,secmin;
    int secmin;
    char line[MAXLINE];
    char shortest[MAXLINE];
    char secshortest[MAXLINE];

    min = 1000;
    secmin=1000;
    while ((len = readline(line, MAXLINE)) > 0)   //用len来保存行长度
    {
        printf ("%d, %s", len, line);  //相比上个程序，这里增加了一行打印
        if (len < min&&secmin>min)
        {
            secmin=min;
            min = len;
            copy (shortest, line);
            copy (secshortest, shortest);
        }
    }
    if (min > 0)
        printf ("第二短文本行为%s", secshortest);
    	printf ("最短文本行为%s", shortest);
    
    /*以上部分为打印最短文本行,以下为打印第二短文本行*/
    
    return 0;
}

int readline(char s[], int lim)
{
    int c, i, j; //增加一个变量j

    j = 0;

    for (i = 0; (c = getchar()) != EOF && c != '\n'; ++i)
    {
        if (i < lim - 2)   //最多存储lim-2个字符到s数组中，如大于该数值则只是增加变量i的值用于计算该行长度
        {
            s[j] = c;
            ++j;
        }
    }

    if (c == '\n')  //如果该条件成立说明上面for循环执行语句没执行，需要将换行符加上
    {
        s[j] = c;
        ++j;
        ++i;
    }

    s[j] = '\0';
    return i;
}

void copy (char to[], char from[])
{
    int i;

    i = 0;
    while ((to[i] = from[i]) != '\0')  //复制字符串
        i++;
}

第五题

函数replacePart(int x,int y, int p,int len)

作用为将x的第p位之后的len个二进制位替换成y变量的最低len位，作为函数返回值返回。

请实现该函数，并且编写主程序，读入x,y,p,len，输出替换后的结果

#include <stdio.h>

unsigned replace(unsigned x, int p, int n, unsigned y);

int main()
{
	printf("输入被替换数x,需要被替换的位数p,替换用的数y，替换用的位数len\n");
    
	int pos,cnt;		
    unsigned int num,numset;
    scanf("%u",&num);
    scanf("%d",&pos);
    scanf("%u",&numset);
    scanf("%d",&cnt);
    
    printf("替换后的数为%u\n", replace(num, pos, cnt, numset));
	

	return 0;
}

/*函数名称：setbits
  功能：将x中从第p位开始的n个位设置成y中最右边n位的值，
        x的其余各位保持不变
  参数：x：待处理的数
		p：从y中获取的字段放入x中的开始位置
		n：从y中获取的字段的长度
		y：被获取字段的数
  返回值：x中第p位开始长度为n的二进制位被替代成y的最右n个字段的值，其他位保持不变，然后返回这个结果。
*/

unsigned replace(unsigned x, int p, int n, unsigned y)
{
	int tmp;

	tmp = (~0 << p+1) | ~(~0 << p+1-n);	//tmp：从第p位开始，长度为n，其中位值为0，其余
	                                    //位值为1的二进制数--此处用十进制数来表示。
	x = x & tmp;						//将x中的第p位开始长度为n的位值置为0，其余不变。
	y = (y << p+1-n) & ~tmp;			/*将y左移p+1-n位，然后与~tmp相与，这样y的第p位开始，长度为n的位的值即为操作前的y的最右n位的值，而此时y的其余位的值为0。*/
	return x | y;		//将操作后的x,y相与，即实现功能。

}
